package backend

import (
        "flag"
        "fmt"
        "github.com/PatrickLaabs/eros/pkg/backend/routes"
        "html"
        "log"
        "net/http"
        "strings"
)

/*
Backend api server.
Provides APIs for the frontend to consume, as creating and managing kubernetes clusters.
*/

var (
        greeting = flag.String("g", "Hello", "Greet with `greeting`")
        addr     = flag.String("addr", ":8080", "address to server")
)

func greet(w http.ResponseWriter, r *http.Request) {
        name := strings.Trim(r.URL.Path, "/")
        if name == "" {
                name = "Gopher"
        }

        _, err := fmt.Fprintf(w, "%s, %s!\n", *greeting, html.EscapeString(name))
        if err != nil {
                return
        }
}

func Start() {
        http.HandleFunc("/", greet)
        http.HandleFunc("/version", routes.Version)

        log.Fatal(http.ListenAndServe(*addr, nil))
}
